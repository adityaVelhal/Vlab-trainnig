Sum of Digits of Numbers

Code

#include <stdio.h>
int main() {
int num=0, sum=0, remainder=0;
printf(" Enter a number : ");
scanf("%d", &num);
while(num>0) {
remainder = num%10;
sum = sum + remainder;
num = num/10;
}
printf("Sum of digits in number : %d", sum);
}
